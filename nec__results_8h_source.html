125  inline void section_start(const std::string& section_name)
126  {
127  if (m_in_section)
128  section_end();

130  switch (m_format)

139  break;

141  default:

143  break;
144  }
145  m_in_section = true;

423  os << " NORMALIZATION FACTOR: ";oh.real_out(11,4,norm_factor);os << endl;
424  os << " ETA: ";oh.real_out(7,2,_eta,false); os << " DEGREES" << endl;
425  os << " TYPE: " << _type << endl;
426  os << " AXIAL RATIO: "; oh.real_out(6,3,_axial_ratio,false); os << endl;
427  os << " SEGMENT No: ";oh.int_out( 5, _segment_number); os << endl << endl;
428  os << " THETA PHI ---- PATTERN ----" << endl;
429  os << " (DEG) (DEG) DB MAGNITUDE" << endl;
430 
431  nec_float theta = _theta0;
432 
433  for (int t=0; t<n_theta; t++)
434  {
435  nec_float phi = _phi0;
436 
437  for (int p=0; p<n_phi;p++)
438  {
439  nec_float magnitude = _mag(t,p) / norm_factor;
440  nec_float gain = db20(magnitude);
441 
442  oh.start_record();
443  oh.padding(" ");
444  oh.real_out(7,2, theta, false); oh.separator();
445  oh.real_out(7,2, phi, false); oh.separator();
446  oh.padding(" "); oh.real_out(7,2, gain, false); oh.separator();
447  oh.padding(" "); oh.real_out(11,4, magnitude);
448  oh.end_record();
449 
450  phi += _phi_step;
451  }
452  theta += _theta_step;
453  }
454  }
